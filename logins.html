function validateForm() { var name = document.getElementById("name").value; var email = document.getElementById("email").value; if (name === "" || email === "") { alert("All fields are required!"); return false; } var emailPattern = /^[^ ]+@[^ ]+\.[a-z]{2,3}$/; if (!email.match(emailPattern)) { alert("Enter a valid email address!"); return false; } alert("Form submitted successfully!"); return true; }
